On system#boot do
  // timerset,1,60 // time
  gpio,2,0 // blue LED ON
endon

On rules#timer=1 do
  timerset,1,60
  publish %sysname%/status,"%sysname% IP = %ip%"
  Publish %sysname%/time/value,%systime%
  publish %sysname%/a0/a0,[a0#a0]
endon

on Wifi#Disconnected do 
// TODO set disconnect counter
// and goto BED
    gpio,2,1 // blue LED OFF
    deepsleep,3600
endon

On MQTT#connected do
    publish %sysname%/status,"%systime%: %sysname%/%ip% %rssi% dBm connected, MAC is %bssid%."
    Publish %sysname%/time/value,%systime%
//    publish %sysname%/bmp085/press,[bmp085#press]
//    publish %sysname%/bmp085/temp,[bmp085#temp]
    publish %sysname%/a0/a0,[a0#a0]
    gpio,2,0 // blue LED ON
    // wait 10 secs to give chance to disable sleep mode
    delay,10000
    // enter sleep mode
    timerset,2,1
endon


On MqttListener#sleepMode=1 do
  publish %sysname%/status,'sleepMode of %sysname%/%ip% ACTIVE, please wait 1h  til wakeup'
  publish %sysname%/a0/a0,[a0#a0]
  gpio,2,1
  delay,1000
  pulse,2,0,200
  timerset,2,1
endon

// somebody disabled sleep
On MqttListener#sleepMode=0 do
  timerset,2,0
  gpio,2,0
  publish %sysname%/status,'%sysname%/%ip% sleepMode OFF'
  publish %sysname%/a0/a0,[a0#a0]
endon

On rules#timer=2 do
  publish %sysname%/status,'%systime%: %sysname%/%ip% go to BED, %rssi% dBm'
  deepsleep,3600
endon

on Clock#Time=All,05:52 do 
  // restart because living room access point restarts at 05:39 a.m.
  reboot
endon
